    /**
     * 设置是否使用HTTPS协议
     * @param bool $bolUseHttps 当该值为true时，设置curl为支持https协议
     */
    public function setUseHttps($bolUseHttps = true) {
        if ($bolUseHttps) {
            // 对认证证书来源的检查
            curl_setopt($this->objCurl, CURLOPT_SSL_VERIFYPEER, 0);
            // 从证书中检查SSL加密算法是否存在
            curl_setopt($this->objCurl, CURLOPT_SSL_VERIFYHOST, 1);
        }
    }
    
    public function __destruct() {
        curl_close($this->objCurl);
    }
    
    /**
     * 向指定url发送post请求
     * @param string $strUrl 要发送post请求的url
     * @param string $arrData 要post的数据
     */
    public function post($strUrl, $arrData) {
        curl_setopt($this->objCurl, CURLOPT_POST, 1);
        curl_setopt($this->objCurl, CURLOPT_URL, $strUrl);
        curl_setopt($this->objCurl, CURLOPT_POSTFIELDS, $arrData);
        
        $arrOutput = curl_exec($this->objCurl);
        if (!$arrOutput) {
            trigger_error(curl_error($this->objCurl));
        }
        return json_decode($arrOutput);
    }
    
    /**
     * 向指定url发送GET请求
     * @param string $strUrl 接收请求的url
     */
    public function get($strUrl) {
        curl_setopt($this->objCurl, CURLOPT_HTTPGET, 1);
        curl_setopt($this->objCurl, CURLOPT_URL, $strUrl);
        
        $arrOutput = curl_exec($this->objCurl);
        if (!$arrOutput) {
            trigger_error(curl_error($this->objCurl));
        }
        return $arrOutput;
    }
